    <script>
        // Use DOMContentLoaded to ensure elements exist before referencing them
        document.addEventListener('DOMContentLoaded', () => {
            
            // Initialize Lucide Icons
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }

            // Element References
            const widthInput = document.getElementById('widthInput');
            const depthInput = document.getElementById('depthInput');
            const priceInput = document.getElementById('priceInput');
            
            const boothShape = document.getElementById('boothShape');
            const container = document.getElementById('visualizerContainer');
            
            const visualArea = document.getElementById('visualArea');
            const resultArea = document.getElementById('resultArea');
            const resultCost = document.getElementById('resultCost');
            
            const textWidth = document.getElementById('textWidth');
            const textDepth = document.getElementById('textDepth');
            const textPrice = document.getElementById('textPrice');
            const displayDims = document.getElementById('displayDims');
            const widthLabel = document.getElementById('widthLabel');
            const depthLabel = document.getElementById('depthLabel');

            // State
            let currentCost = 0;

            // Functions defined inside the listener to access elements
            
            // Attached to window so HTML onlick attributes can find them
            window.calculate = function() {
                let width = 0;
                let depth = 0;
                let price = 0;

                if (widthInput) width = parseFloat(widthInput.value) || 0;
                if (depthInput) depth = parseFloat(depthInput.value) || 0;
                if (priceInput) price = parseFloat(priceInput.value) || 0;

                // Clamping for sanity
                if (width < 0) width = 0;
                if (depth < 0) depth = 0;

                const area = width * depth;
                const cost = area * price;

                // Update Text with Safe Checks
                if (visualArea) visualArea.textContent = area.toLocaleString();
                if (resultArea) resultArea.textContent = area.toLocaleString();
                
                if (resultCost) {
                    animateValue(resultCost, currentCost, cost, 500);
                    currentCost = cost;
                }

                if (textWidth) textWidth.textContent = width;
                if (textDepth) textDepth.textContent = depth;
                if (textPrice) textPrice.textContent = price.toFixed(2);
                if (displayDims) displayDims.textContent = `${width}' x ${depth}'`;
                if (widthLabel) widthLabel.textContent = `${width}'`;
                if (depthLabel) depthLabel.textContent = `${depth}'`;

                updateVisualizer(width, depth);
            }

            function updateVisualizer(width, depth) {
                if (!container || !boothShape) return;

                // Get container dimensions
                const containerW = container.clientWidth;
                const containerH = container.clientHeight;
                const padding = 70; // Increased padding for labels

                const availW = containerW - padding;
                const availH = containerH - padding;
                
                // Calculate scale
                let scale = Math.min(availW / width, availH / depth);

                // Cap the scale
                const maxScale = 25; 
                if (scale > maxScale) scale = maxScale;

                // If dimensions are 0
                if (width === 0 || depth === 0) scale = 0;

                const finalW = width * scale;
                const finalH = depth * scale;

                // Apply styles
                boothShape.style.width = `${finalW}px`;
                boothShape.style.height = `${finalH}px`;
            }

            // Animated Number Counter
            function animateValue(obj, start, end, duration) {
                if (!obj) return;
                let startTimestamp = null;
                const step = (timestamp) => {
                    if (!startTimestamp) startTimestamp = timestamp;
                    const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                    const ease = 1 - Math.pow(1 - progress, 4);
                    const currentVal = start + (end - start) * ease;
                    
                    obj.innerHTML = new Intl.NumberFormat('en-US', {
                        style: 'currency',
                        currency: 'USD'
                    }).format(currentVal);

                    if (progress < 1) {
                        window.requestAnimationFrame(step);
                    }
                };
                window.requestAnimationFrame(step);
            }

            // Preset Handler - Attached to window
            window.setPreset = function(w, d) {
                if (widthInput) widthInput.value = w;
                if (depthInput) depthInput.value = d;
                
                // Trigger animation effect on inputs
                if (widthInput && widthInput.parentElement) {
                    widthInput.parentElement.classList.add('scale-105');
                    setTimeout(() => widthInput.parentElement.classList.remove('scale-105'), 150);
                }
                
                window.calculate();
                createParticles();
            };

            // Clear All Function - Attached to window
            window.clearAll = function() {
                if (widthInput) widthInput.value = 0;
                if (depthInput) depthInput.value = 0;
                if (priceInput) priceInput.value = 51.50;
                
                window.calculate();
                
                if (resultCost) {
                    const costCard = resultCost.closest('.glass-panel');
                    if (costCard) {
                        costCard.classList.add('brightness-150');
                        setTimeout(() => costCard.classList.remove('brightness-150'), 200);
                    }
                }
            };

            // Click Effect (Particles)
            function createParticles() {
                if (!boothShape) return;
                const rect = boothShape.getBoundingClientRect();

                for (let i = 0; i < 5; i++) {
                    const particle = document.createElement('div');
                    particle.classList.add('particle');
                    const size = Math.random() * 6 + 2;
                    particle.style.width = `${size}px`;
                    particle.style.height = `${size}px`;
                    particle.style.left = `${(rect.width/2) + (Math.random() * 40 - 20)}px`;
                    particle.style.top = `${(rect.height/2) + (Math.random() * 40 - 20)}px`;
                    boothShape.appendChild(particle);
                    setTimeout(() => particle.remove(), 2000);
                }
            }

            // Listeners
            if (widthInput) widthInput.addEventListener('input', window.calculate);
            if (depthInput) depthInput.addEventListener('input', window.calculate);
            if (priceInput) priceInput.addEventListener('input', window.calculate);
            window.addEventListener('resize', window.calculate);

            // Initial Run
            window.calculate();

        });
    </script>
